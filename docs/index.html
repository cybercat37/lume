The Forge

Every program is shaped through a clear, explicit pipeline. Each step does one job, and does it well.

Parse

Turn source into syntax. Preserve structure and intent.

Bind

Resolve symbols, types, and diagnostics early.

Lower

Rewrite into a simpler executable core.

Emit / Run

Interpret or compile to clean, readable C#.

Channels with explicit recv handling

scope {
  let (tx, rx) = channel()

  let sender = spawn { tx.send(42) }
  let worker = spawn {
    print match rx.recv() {
      Ok(value) -> value
      Error(msg) -> -1
    }
  }

  sender.join()
  worker.join()
}

Range, enumerate, zip_with

let xs = range(1, 6)
let indexed = enumerate(xs)
let mixed = zip_with(xs, [10, 20, 30, 40, 50],
  fn(x: Int, y: Int) => x + y)

print indexed
print mixed |> fold(0, fn(acc: Int, x: Int) => acc + x)

Interpolation + formatting

let name = "ada"
let n = 7
let price = 12.5

print f"Hello, {name:upper}!"
print f"n={n:000}"
print f"price={price:0.00}"
print f"literal braces: {{ok}}"

Result-friendly random API

rand_seed(7)
let value = rand_int(10)

print match value {
  Ok(n) -> n
  Error(msg) -> -1
}

Records and updates

type User { name: String, age: Int }

let user = User { name: "Ada", age: 36 }
let updated = user with { age: 37 }

print user.name
print updated.age
